C++ I: Page 2 - Variables & Primitives

The Building Blocks of Data

C++ is statically typed, meaning you must declare the type of a variable before using it. This allows the compiler to perform rigorous checks and optimize memory layout for specific data sizes.

The Code implementation


/*
 * File: primitives.cpp
 * Demonstration of C++ primitive types and type modifiers
 */
#include <iostream>
#include <string>
#include <limits> // For numerical limits

using namespace std;

// Enum for readable state constants
enum State { IDLE, RUNNING, ERROR };

// Struct for grouping related primitives
struct UserProfile {
    int id;
    double balance;
    bool isActive;
};

int main() {
    // Integer types with modifiers
    short s = 10;            // Usually 2 bytes
    int i = 1000;            // Usually 4 bytes
    long l = 1000000L;       // Usually 4 or 8 bytes
    unsigned int ui = 500U;  // Only non-negative

    // Floating point
    float f = 3.14f;
    double d = 3.1415926535;

    // Characters and Strings
    char c = 'A';
    string str = "Hello C++";

    // Boolean
    bool isRunning = true;

    // Constants (Cannot be modified)
    const int MAX_USERS = 100;
    
    cout << "--- Primitive Types Summary ---" << endl;
    cout << "Integer: " << i << endl;
    cout << "Double: " << d << endl;
    cout << "String: " << str << endl;
    cout << "Is Running: " << (isRunning ? "Yes" : "No") << endl;

    // Numerical limits
    cout << "Min int: " << numeric_limits<int>::min() << endl;
    cout << "Max int: " << numeric_limits<int>::max() << endl;

    // Using the struct
    UserProfile user = {1, 150.50, true};
    cout << "User ID: " << user.id << " | Balance: " << user.balance << endl;

    return 0;
}
