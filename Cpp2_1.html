Dynamic Arrays with std::vector

The std::vector is the workhorse of modern C++ containers. It manages dynamic arrays, automatically handling memory resizing as elements are added or removed.

The Code implementation


/*
 * File: vectors.cpp
 * Demonstration of std::vector functionality
 */
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 1. Creation and Initialization
    vector<int> numbers = {10, 20, 30};
    
    // 2. Dynamic resizing
    numbers.push_back(40);
    numbers.push_back(50);
    
    // 3. Access
    cout << "First element: " << numbers[0] << endl;
    cout << "Size: " << numbers.size() << endl;
    
    // 4. Modification
    numbers.pop_back(); // Removes 50
    
    // 5. Iteration
    cout << "All elements: ";
    for (int n : numbers) {
        cout << n << " ";
    }
    cout << endl;

    // 6. STL Algorithms with Vector
    sort(numbers.begin(), numbers.end(), greater<int>());
    
    cout << "Sorted (desc): ";
    for (int n : numbers) {
        cout << n << " ";
    }
    cout << endl;
    
    return 0;
}

/*
 * Vector Nuances:
 * 1. Vectors automatically manage heap memory.
 * 2. 'size()' returns current elements, 'capacity()' returns total space allocated.
 * 3. Vectors resize themselves when capacity is exceeded, which can be costly (O(n)).
 * 4. Use 'reserve()' if you know the final size to prevent multiple reallocations.
 * 5. Accessing out of bounds using '[]' causes undefined behavior; use '.at()' for safety.
